21 void DefaultAnalyzer::run() {
22  auto p = filesystem::path(path);
23 
24  if (!filesystem::exists(path))
25  {
26  status = "fail";
27  message.append(path + " does not exists. \n");
28  throw(ParserException(message));
29  }
30 
31  filename = p.filename();
32  ext = p.extension();
33  filesize = filesystem::file_size(p);
34 
35  // Try open record file
36  _f.open(path, ifstream::in | ifstream::binary);
37  if (!_f.is_open())
38  {
39  status = "fail";
40  message.append("Failed to open " + path + ". \n");
41  throw(ParserException(message));
42  }
43 
44  // Try to extract header&body streams
45  if (!_locateStreams())
46  {
47  status = "fail";
48  message.append("Failed to unzip raw header data. \n");
49  throw(ParserException(message));

53 
54  // Log current stage
55  status = "good";
56  message.append("[INFO] Successfully extracted header&body streams from this file. Ready for data analyzing. \n");

62 bool DefaultAnalyzer::_locateStreams() {
63  /**
64  * \brief headerMeta[0]: header_len + next_pos + header_data (Data
65  * length) \n headerMeta[1]: nextPos
66  *
67  */
68  int32_t headerMeta[2];
69  _f.read((char*)headerMeta, 8);
70 
71  // Construct body stream
72  _bodySize = filesize - headerMeta[0];
73  _body.resize(_bodySize); /// \note vector 的 resize 和 reserve 是不一样的，这里因为这个问题卡了很久。reserve 并不会初始化空间，因此也不能直接读数据进去。
74  _f.seekg(headerMeta[0]);
75  _f.read((char*)_body.data(), _bodySize);
76 
77  uintmax_t rawHeaderPos = headerMeta[1] < filesize ? 8 : 4;
78  _f.seekg(rawHeaderPos);
79  if (rawHeaderPos == 4) versionCode = AOK;
80 
81  // Try to unzip header data
82  if (_inflateRawHeader() != 0) return false;
83 
84  return true;
85 }
