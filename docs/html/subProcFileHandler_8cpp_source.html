44 int DefaultAnalyzer::_inflateRawHeader()
45 {
46  int ret;
47  unsigned have;
48  z_stream strm;
49  uint8_t in[ZLIB_CHUNK];
50  uint8_t out[ZLIB_CHUNK];
51 
52  _header.reserve(HEADER_INIT);
53 
54  /* allocate inflate state */
55  strm.zalloc = Z_NULL;
56  strm.zfree = Z_NULL;
57  strm.opaque = Z_NULL;
58  strm.avail_in = 0;
59  strm.next_in = Z_NULL;
60  ret = inflateInit2(&strm, -MAX_WBITS);
61  if (ret != Z_OK)
62  return ret;
63 
64  /* decompress until deflate stream ends or end of file */
65  do
66  {
67  if (FILE_INPUT == _inputType)
68  {
69  _f.read((char *)&in, ZLIB_CHUNK);
70  strm.avail_in = _f.gcount();
71  if (!_f.good())
72  {
73  (void)inflateEnd(&strm);
74  return Z_ERRNO;
75  }
76  }
77  else
78  {
79  strm.avail_in = filesize - (_curPos - _b) >= ZLIB_CHUNK ? ZLIB_CHUNK : filesize - (_curPos - _b);
80  memcpy(&in, _curPos, strm.avail_in);
81  _curPos += strm.avail_in;
82  }
83 
84  if (strm.avail_in == 0)
85  break;
86  strm.next_in = in;
87 
88  /* run inflate() on input until output buffer not full */
89  do
90  {
91  strm.avail_out = ZLIB_CHUNK;
92  strm.next_out = out;
93 
94  ret = inflate(&strm, Z_NO_FLUSH);
95  // assert(ret != Z_STREAM_ERROR); /* state not clobbered */
96  switch (ret)
97  {
98  case Z_NEED_DICT:
99  ret = Z_DATA_ERROR; /* and fall through */
100  case Z_DATA_ERROR:
101  case Z_MEM_ERROR:
102  (void)inflateEnd(&strm);
103  return ret;
104  }
105 
106  have = ZLIB_CHUNK - strm.avail_out;
107  _header.insert(
108  _header.end(),
109  out,
110  out + have);
111  if (!_f.good() && (FILE_INPUT == _inputType))
112  {
113  (void)inflateEnd(&strm);
114  return Z_ERRNO;
115  }
116  } while (strm.avail_out == 0);
117 
118  /* done when inflate() says it's done */
119  } while (ret != Z_STREAM_END);
120 
121  /* clean up and return */
122  (void)inflateEnd(&strm);
123  return ret == Z_STREAM_END ? Z_OK : Z_DATA_ERROR;
124 }
