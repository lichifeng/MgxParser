18  unsigned have;

21  uint8_t out[ZLIB_CHUNK];

26  strm.zalloc = Z_NULL;
27  strm.zfree = Z_NULL;
28  strm.opaque = Z_NULL;
29  strm.avail_in = 0;
30  strm.next_in = Z_NULL;
31  ret = inflateInit2(&strm, -MAX_WBITS);
32  if (ret != Z_OK)
33  return ret;
34 
35  /* decompress until deflate stream ends or end of file */
36  do
37  {
38  _f.read((char *)&in, ZLIB_CHUNK);
39  strm.avail_in = _f.gcount();
40  if (!_f.good())
41  {
42  (void)inflateEnd(&strm);
43  return Z_ERRNO;
44  }
45  if (strm.avail_in == 0)
46  break;
47  strm.next_in = in;
48 
49  /* run inflate() on input until output buffer not full */
50  do
51  {
52  strm.avail_out = ZLIB_CHUNK;
53  strm.next_out = out;
54 
55  ret = inflate(&strm, Z_NO_FLUSH);
56  // assert(ret != Z_STREAM_ERROR); /* state not clobbered */
57  switch (ret)
58  {
59  case Z_NEED_DICT:
60  ret = Z_DATA_ERROR; /* and fall through */
61  case Z_DATA_ERROR:
62  case Z_MEM_ERROR:
63  (void)inflateEnd(&strm);
64  return ret;
65  }
66 
67  have = ZLIB_CHUNK - strm.avail_out;
68  _header.insert(
69  _header.end(),
70  out,
71  out + have);
72  if (!_f.good())
73  {
74  (void)inflateEnd(&strm);
75  return Z_ERRNO;
76  }
77  } while (strm.avail_out == 0);
78 
79  /* done when inflate() says it's done */
80  } while (ret != Z_STREAM_END);
81 
82  /* clean up and return */
83  (void)inflateEnd(&strm);
84  return ret == Z_STREAM_END ? Z_OK : Z_DATA_ERROR;
85 }
