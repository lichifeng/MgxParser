31 void DefaultAnalyzer::_guessWinner(int debugFlag)
32 {
33  _debugFlag = debugFlag;
34 
35  multimap<uint8_t, TeamCredit> winnerCredits;
36  uint32_t indexMax[9] = {0};
37  uint8_t indexTeam[9] = {0};
38  TeamCredit tc;
39  uint32_t credit = 0, creditMax = 0;
40  map<uint8_t, TeamCredit>::iterator found;
41  bool allLived = true; // When nobody had resigned, that mean POV quit first, then POV is possibly lost.
42 
43  for (auto &p : players)
44  {
45  if (!p.valid())
46  continue;
47 
48  if (p.index < 0 || p.index > 8)
49  {
50  logger->warn("Flag:{}, bad player index:{}! Name: {}", _debugFlag, p.index, p.name);
51  }
52  if (-1 == p.resigned)
53  {
54  credit = duration + 500; // \note 有时候投降时间会和时长一致，那就把没投降的人分数多给一点点，这样就不会判断为已经投降的和没投降都是赢。
55  }
56  else
57  {
58  credit = p.resigned;
59  allLived = false;
60  }
61  p.resolvedTeamID = 1 == p.resolvedTeamID ? 10 + p.index : p.resolvedTeamID;
62  indexTeam[p.index] = p.resolvedTeamID;
63  if (credit > indexMax[p.index])
64  indexMax[p.index] = credit;
65  }
66 
67  for (int i = 0; i < 9; i++)
68  {
69  if (0 == indexTeam[i])
70  continue;
71 
72  found = winnerCredits.find(indexTeam[i]);
73  if (found == winnerCredits.end())
74  {
75  tc.credits = indexMax[i];
76  tc.count = 1;
77  winnerCredits.emplace(make_pair(indexTeam[i], tc));
78  }
79  else
80  {
81  found->second.credits += indexMax[i];
82  found->second.count++;
83  }
84  }
85 
86  for (auto &t : winnerCredits)
87  {
88  if (teamMode.empty())
89  teamMode.append(to_string(t.second.count));
90  else
91  teamMode.append("v").append(to_string(t.second.count));
92 
93  t.second.avg = t.second.credits / t.second.count;
94  if (t.second.avg > creditMax)
95  creditMax = t.second.avg;
96  }
97 
98  for (auto &p : players)
99  {
100  if (!p.valid())
101  continue;
102 
103  found = winnerCredits.find(p.resolvedTeamID);
104  if (found != winnerCredits.end() && found->second.avg == creditMax && !(p.slot == recPlayer && allLived))
105  p.isWinner = true;
106  }
107 }
