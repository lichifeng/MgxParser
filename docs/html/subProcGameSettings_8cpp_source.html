49  {
50  namePtr = _curPos + 8;
51  // 先获得名字长度，再把 len + lenName + trailBytes 放到 pattern 里，
52  // 供以后搜索使用， 然后再跳过或把名字转码保存下来。
53  // trailBytes 只保存了一个指针，具体用后面的几个字节组装可以试下，看
54  // 看性能表现，一般建议在[1, 6]个，第一个一般是0x16,第六个一般是
55  // 0x21，中间一个4字节的数字则根据版本不同有所区别。
56  // 注1：HD/DE版本中，startinfo里的名字数据和HD/DE header里的一
57  // 样，和这里不一样。这里显示的玩家名是“玩家1”之类。
58  // 注2：startinfo中字符串长度用2字节表示，这里用4字节。
59  // \todo 这里似乎只有玩家才会在HD/DE专有区块有名字信息，GAIA和AI没有。
60  // 注3：startinfo中的字符串有\0结尾，这里没有
61 
62  _readBytes(4, &players[i].slot);
63  if (players[i].name.length() > 0)
64  {
65  // Data from HD/DE header has higher priority
66  _skip(4);
67  nameLen = players[i].name.length() + 1;
68  players[i].searchPattern.resize(2 + nameLen + trailBytes);
69  memcpy(players[i].searchPattern.data(), &nameLen, 2);
70  memcpy(players[i].searchPattern.data() + 2, players[i].name.data(), nameLen);
71  memcpy(players[i].searchPattern.data() + 2 + nameLen, _startInfoPatternTrail, trailBytes);
72  _skipPascalString(true);
73  }
74  else
75  {
76  _readBytes(4, &players[i].type);
77  // 两处字符串前面有\0，后面有的版本有，有的版本没有，需要判断处理
78  nameLen = *(uint32_t *)namePtr;
79  if ('\0' == *(namePtr + 4 + nameLen - 1))
80  {
81  players[i].searchPattern.resize(2 + nameLen + trailBytes);
82  memcpy(players[i].searchPattern.data() + 2, namePtr + 4, nameLen);
83  }
84  else
85  {
86  nameLen += 1;
87  players[i].searchPattern.resize(2 + nameLen + trailBytes);
88  players[i].searchPattern[2 + nameLen - 1] = '\0';
89  memcpy(players[i].searchPattern.data() + 2, namePtr + 4, nameLen - 1);
90  }
91  memcpy(players[i].searchPattern.data(), &nameLen, 2);
92  memcpy(players[i].searchPattern.data() + 2 + nameLen, _startInfoPatternTrail, trailBytes);
93  _readPascalString(players[i].name, true, true);
94  }
95  }
