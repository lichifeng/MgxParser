15 void DefaultAnalyzer::_findInitialPlayersDataPos() {
16  _curPos = _startInfoPos + 2 + numPlayers + 36 + 4 + 1;
17 
18  uint32_t easySkip = easySkipBase + mapCoord[0] * mapCoord[1];
19  auto itStart = _header.cbegin() + (_curPos - _header.data()) + easySkip;
20  auto itEnd = _header.cbegin() + (_scenarioHeaderPos - _header.data() - numPlayers * 1817); // Achievement section is 1817 * numPlayers bytes
21 
22  // Length of every player's data is at least easySkip bytes (and ususally much
23  // more), we can use this to escape unnecessary search
24  // First player don't need a search (and cannot, 'cuz different behavior
25  // among versions)
26  players[0].dataOffset = _curPos - _header.data();
27 
28  auto found = itStart;
29  for (size_t i = 1; i < numPlayers; i++)
30  {
31  found = findPosition(
32  itStart, itEnd,
33  players[i].searchPattern.cbegin(),
34  players[i].searchPattern.cend()
35  );
36  if (found == itEnd) {
37  cout << "now at: " << i << " " << players[i].name << endl;;
38  throw(ParserException("[WARN] Cannot find satisfied player data in startinfo. \n"));
39  }
40  players[i].dataOffset = found - _header.cbegin() - (2 + numPlayers + 36 + 4 + 1);
41  itStart = found + easySkip;
42  }
43  message.append("[INFO] Located player initial data position in start info section. \n");
44 }
