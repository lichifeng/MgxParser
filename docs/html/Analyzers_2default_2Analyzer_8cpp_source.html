22 void DefaultAnalyzer::run() {
23  auto p = filesystem::path(path);
24 
25  if (!filesystem::exists(path))
26  {
27  status = "fail";

31

35 
36  // Try open record file
37  _f.open(path, ifstream::in | ifstream::binary);
38  if (!_f.is_open())
39  {
40  status = "fail";

42  throw(AnalyzerException(message));

44 
45  // Try to extract header&body streams

48  status = "fail";

63 bool DefaultAnalyzer::_locateStreams() {
64  /**
65  * \brief headerMeta[0]: header_len + next_pos + header_data (Data
66  * length) \n headerMeta[1]: nextPos
67  *
68  */
69  int32_t headerMeta[2];
70  _f.read((char*)headerMeta, 8);
71 
72  // Construct body stream
73  _bodySize = filesize - headerMeta[0];
74  _body.resize(_bodySize); /// \note vector 的 resize 和 reserve 是不一样的，这里因为这个问题卡了很久。reserve 并不会初始化空间，因此也不能直接读数据进去。
75  _f.seekg(headerMeta[0]);
76  _f.read((char*)_body.data(), _bodySize);
77 
78  uintmax_t rawHeaderPos = headerMeta[1] < filesize ? 8 : 4;
79  _f.seekg(rawHeaderPos);
80  if (rawHeaderPos == 4) versionCode = AOK;
81 
82  // Try to unzip header data
83  if (_inflateRawHeader() != 0) return false;
84 
85  return true;
86 }

131 void DefaultAnalyzer::_mapDataAnalyzer() {

140  }
141 
142  int32_t numMapZones, mapBits, numFloats;
143  mapBits = mapCoord[0] * mapCoord[1];
144  _readBytes(4, &numMapZones);
145  for (int i = 0; i < numMapZones; i++)
146  {
147  if (IS_HD(versionCode) || IS_DE(versionCode)) /// \todo 为什么不是11.76？
148  _skip(2048 + mapBits * 2);
149  else
150  _skip(1275 + mapBits);
151  _readBytes(4, &numFloats);
152  _skip(numFloats * 4 + 4);
153  }
154 
155  _readBytes(1, &allVisible);
156  _readBytes(1, &fogOfWar);
157 
158  _mapBitmap = _curPos;
159  uint32_t checkVal = *(uint32_t*)(_curPos + 7 * mapBits);
160  if (IS_DE(versionCode)) {
161  _mapTileType = (saveVersion >= 13.0299 || checkVal > 1000) ? 9 : 7;
162  } else {
163  _mapTileType = (_curPos[0] == 255) ? 4 : 2;
164  }
165  _skip(_mapTileType * mapBits);
166 
167  int32_t numData, numObstructions;
168  _readBytes(4, &numData);
169  _skip(4 + numData * 4);
170  for (int i = 0; i < numData; i++)
171  {
172  _readBytes(4, &numObstructions);
173  _skip(numObstructions * 8);
174  }
175  int32_t visibilityMapSize[2];

177  _skip(visibilityMapSize[0] * visibilityMapSize[1] * 4);

627 int DefaultAnalyzer::_inflateRawHeader()
628 {
629  int ret;
630  unsigned have;
631  z_stream strm;
632  uint8_t in[ZLIB_CHUNK];
633  uint8_t out[ZLIB_CHUNK];
634 
635  _header.reserve(HEADER_INIT);
636 
637  /* allocate inflate state */
638  strm.zalloc = Z_NULL;
639  strm.zfree = Z_NULL;
640  strm.opaque = Z_NULL;
641  strm.avail_in = 0;
642  strm.next_in = Z_NULL;
643  ret = inflateInit2(&strm, -MAX_WBITS);
644  if (ret != Z_OK)
645  return ret;
646 
647  /* decompress until deflate stream ends or end of file */
648  do {
649  _f.read((char*)&in, ZLIB_CHUNK);
650  strm.avail_in = _f.gcount();
651  if (!_f.good()) {
652  (void)inflateEnd(&strm);
653  return Z_ERRNO;
654  }
655  if (strm.avail_in == 0)
656  break;
657  strm.next_in = in;
658 
659  /* run inflate() on input until output buffer not full */
660  do {
661  strm.avail_out = ZLIB_CHUNK;
662  strm.next_out = out;
663 
664  ret = inflate(&strm, Z_NO_FLUSH);
665  // assert(ret != Z_STREAM_ERROR); /* state not clobbered */
666  switch (ret) {
667  case Z_NEED_DICT:
668  ret = Z_DATA_ERROR; /* and fall through */
669  case Z_DATA_ERROR:
670  case Z_MEM_ERROR:
671  (void)inflateEnd(&strm);
672  return ret;
673  }
674 
675  have = ZLIB_CHUNK - strm.avail_out;
676  _header.insert(
677  _header.end(),
678  out,
679  out + have
680  );
681  if (!_f.good()) {
682  (void)inflateEnd(&strm);
683  return Z_ERRNO;
684  }
685  } while (strm.avail_out == 0);
686 
687  /* done when inflate() says it's done */
688  } while (ret != Z_STREAM_END);
689 
690  /* clean up and return */
691  (void)inflateEnd(&strm);
692  return ret == Z_STREAM_END ? Z_OK : Z_DATA_ERROR;
693 }
