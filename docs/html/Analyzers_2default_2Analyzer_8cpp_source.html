22 void DefaultAnalyzer::run() {
23  auto p = filesystem::path(path);
24 
25  if (!filesystem::exists(path))
26  {
27  status = "fail";
28  message.append(path + " does not exists. \n");
29  throw(AnalyzerException(message));
30  }
31 
32  filename = p.filename();
33  ext = p.extension();
34  filesize = filesystem::file_size(p);
35 
36  // Try open record file
37  _f.open(path, ifstream::in | ifstream::binary);
38  if (!_f.is_open())
39  {
40  status = "fail";
41  message.append("Failed to open " + path + ". \n");
42  throw(AnalyzerException(message));
43  }
44 
45  // Try to extract header&body streams
46  if (!_locateStreams())
47  {
48  status = "fail";
49  message.append("Failed to unzip raw header data. \n");
50  throw(AnalyzerException(message));
51  }
52 
53  _f.close();
54 
55  // Log current stage
56  status = "good";
57  message.append("[INFO] Successfully extracted header&body streams from this file. Ready for data analyzing. \n");
58 
59  // Start data analyzing
60  _analyze();
61 }

63 bool DefaultAnalyzer::_locateStreams() {
64  /**
65  * \brief headerMeta[0]: header_len + next_pos + header_data (Data
66  * length) \n headerMeta[1]: nextPos
67  *
68  */
69  int32_t headerMeta[2];
70  _f.read((char*)headerMeta, 8);
71 
72  // Construct body stream
73  _bodySize = filesize - headerMeta[0];
74  _body.resize(_bodySize); /// \note vector 的 resize 和 reserve 是不一样的，这里因为这个问题卡了很久。reserve 并不会初始化空间，因此也不能直接读数据进去。
75  _f.seekg(headerMeta[0]);
76  _f.read((char*)_body.data(), _bodySize);
77 
78  uintmax_t rawHeaderPos = headerMeta[1] < filesize ? 8 : 4;
79  _f.seekg(rawHeaderPos);
80  if (rawHeaderPos == 4) versionCode = AOK;
81 
82  // Try to unzip header data
83  if (_inflateRawHeader() != 0) return false;
84 
85  return true;
86 }

169 void DefaultAnalyzer::_findScenarioHeaderStart() {
170  // Try to locate scenario version data in DE (this float in de varies
171  // among minor game versions and relative stable in previous versions.)
172  if (IS_DE(versionCode)) {
173  size_t searchSpan = 8000; // usually 5500~6500
174  auto startPoint = _victoryStartPos - searchSpan;
175  _curPos = startPoint;
176  for (size_t i = 0; i < searchSpan; i++)
177  {
178  scenarioVersion = *(float*)startPoint;
179  if (scenarioVersion > 1.35 && scenarioVersion < 1.55) {
180  _scenarioHeaderPos = _curPos = startPoint - 4; // 4 bytes are 00 00 00 00 before scenario version
181  message.append("[INFO] Reach _scenarioHeaderPos by scenario version range test. \n");
182  return;
183  }
184  ++startPoint;
185  }
186  throw(AnalyzerException("[WARN] Cannot find satisfied _scenarioHeaderPos in this DE version. \n"));
187  }
188 
189 
190  auto scenarioSeprator = IS_AOK(versionCode) ? \
191  patterns::scenarioConstantAOK : \
192  (
193  IS_HD(versionCode) ?
194  (
195  /// \todo 这里应该是13.3399还是13.3599?
196  saveVersion >= 12.3599 ? \
197  patterns::scenarioConstantHD : \
198  patterns::scenarioConstantMGX2
199  ) :
200  patterns::scenarioConstantAOC
201  );
202 
203  vector<uint8_t>::reverse_iterator rFound;
204  rFound = findPosition(
205  make_reverse_iterator(_header.begin() + (_victoryStartPos - _header.data())),
206  _header.rend(),
207  scenarioSeprator.rbegin(),
208  scenarioSeprator.rend()
209  );
210  if (rFound == _header.rend()) {
211  /// \todo aoc-mgz says aok scenario version can appear in UP
212  throw(AnalyzerException("[WARN] Cannot find satisfied _scenarioHeaderPos. \n"));
213  } else {
214  _scenarioHeaderPos = _curPos = &*(--rFound) - 4 - scenarioSeprator.size();
215  message.append("[INFO] Reach _scenarioHeaderPos and passed validation. \n");
216  }
217 }

288 void DefaultAnalyzer::_findTriggerInfoStart() {
289 
290  // aok ~ hd: 0x9a, 0x99, 0x99, 0x99, 0x99, 0x99, f9, 3f (double 1.6)
291  // de < 13.34: 00 e0 ab 45 + double 2.2
292  // de >= 13.34:
293  // 13.34: 00 e0 ab 45 + padding(1) + double 2.4
294  // 20.06: 00 e0 ab 45 + padding(1) + double 2.4
295  // 20.16: 00 e0 ab 45 + padding(11) + double 2.4
296  // 25.01: 00 e0 ab 45 + padding(11) + double 2.4
297  // 25.02: 00 e0 ab 45 + padding(11) + double 2.4
298  // 25.06: 00 e0 ab 45 + padding(11) + double 2.5
299  // 25.22: 00 e0 ab 45 + padding(11) + double 2.6
300  // 26.16: 00 e0 ab 45 + padding(11) + double 3.0
301  // 26.18: 00 e0 ab 45 + padding(11) + double 3.0
302  // 26.18: 00 e0 ab 45 + padding(11) + double 3.2
303 
304  vector<uint8_t>::reverse_iterator rFound;
305  if (IS_DE(versionCode)) {
306  rFound = findPosition(
307  _header.rbegin(),
308  _header.rend(),
309  patterns::gameSettingSign.rbegin(),
310  patterns::gameSettingSign.rend()
311  );
312  if (rFound == _header.rend()) {
313  throw(AnalyzerException("[WARN] Failed to find _triggerInfoPos (No sign found). \n"));
314  }
315  _curPos = &(*--rFound);
316 
317  if (saveVersion < 13.3399) {
318  if (*(double*)_curPos == 2.2) {
319  _triggerInfoPos = _curPos + 8;
320  } else {
321  throw(AnalyzerException("[WARN] Failed to find _triggerInfoPos. \n"));
322  }
323  }
324  if (saveVersion < 25.0599 && saveVersion >= 13.3399) {
325  if (*(double*)(_curPos + 1) == 2.4) {
326  _triggerInfoPos = _curPos + 1 + 8;
327  } else if (*(double*)(_curPos + 11) == 2.4) {
328  _triggerInfoPos = _curPos + 11 + 8;
329  } else {
330  throw(AnalyzerException("[WARN] Failed to find _triggerInfoPos. \n"));
331  }
332  }
333  if (saveVersion < 25.2199 && saveVersion >= 25.0599) {
334  if (*(double*)(_curPos + 11) == 2.5) {
335  _triggerInfoPos = _curPos + 11 + 8;
336  } else {
337  throw(AnalyzerException("[WARN] Failed to find _triggerInfoPos. \n"));
338  }
339  }
340  if (saveVersion < 26.1599 && saveVersion >= 25.2199) {
341  if (*(double*)(_curPos + 11) == 2.6) {
342  _triggerInfoPos = _curPos + 11 + 8;
343  } else {
344  throw(AnalyzerException("[WARN] Failed to find _triggerInfoPos. \n"));
345  }
346  }
347  if (saveVersion < 26.2099 && saveVersion >= 26.1599) {
348  if (*(double*)(_curPos + 11) == 3.0) {
349  _triggerInfoPos = _curPos + 11 + 8;
350  } else {
351  throw(AnalyzerException("[WARN] Failed to find _triggerInfoPos. \n"));
352  }
353  }
354  if (saveVersion >= 26.2099) {
355  if (*(double*)(_curPos + 11) == 3.2) {
356  _triggerInfoPos = _curPos + 11 + 8;
357  } else if ((*(double*)(_curPos + 11)) > 3.2 && (*(double*)(_curPos + 11)) < 10) {
358  message.append("[WARN] Found unknown but reasonable sign of gameSetting end");
359  _triggerInfoPos = _curPos + 11 + 8;
360  } else {
361  throw(AnalyzerException("[WARN] Failed to find _triggerInfoPos. \n"));
362  }
363  }
364  } else {
365  rFound = findPosition(
366  _header.rbegin(),
367  _header.rend(),
368  patterns::gameSettingSign1.rbegin(),
369  patterns::gameSettingSign1.rend()
370  );
371  if (rFound == _header.rend()) {
372  throw(AnalyzerException("[WARN] Failed to find _triggerInfoPos (No sign found). \n"));
373  }
374  _triggerInfoPos = _curPos = &(*--rFound);
375  }

398 void DefaultAnalyzer::_mapDataAnalyzer() {
399  _readBytes(8, &mapCoord);
400  if (mapCoord[0] >= 10000 || mapCoord[1] >= 10000)
401  {
402  throw(AnalyzerException("[WARN] Abnormal map size data. \n"));
403  } else if (mapCoord[0] == mapCoord[1]) {
404  message.append("[INFO] Found desired map coordinates data. \n");
405  } else {
406  throw(AnalyzerException("[WARN] Map coordinates is weird, X != Y. \n"));
407  }
408 
409  int32_t numMapZones, mapBits, numFloats;
410  mapBits = mapCoord[0] * mapCoord[1];
411  _readBytes(4, &numMapZones);
412  for (int i = 0; i < numMapZones; i++)
413  {
414  if (IS_HD(versionCode) || IS_DE(versionCode)) /// \todo 为什么不是11.76？
415  _skip(2048 + mapBits * 2);
416  else
417  _skip(1275 + mapBits);
418  _readBytes(4, &numFloats);
419  _skip(numFloats * 4 + 4);
420  }
421 
422  _readBytes(1, &allVisible);
423  _readBytes(1, &fogOfWar);
424 
425  _mapBitmap = _curPos;
426  uint32_t checkVal = *(uint32_t*)(_curPos + 7 * mapBits);
427  if (IS_DE(versionCode)) {
428  _mapTileType = (saveVersion >= 13.0299 || checkVal > 1000) ? 9 : 7;
429  } else {
430  _mapTileType = (_curPos[0] == 255) ? 4 : 2;
431  }
432  _skip(_mapTileType * mapBits);
433 
434  int32_t numData, numObstructions;
435  _readBytes(4, &numData);
436  _skip(4 + numData * 4);
437  for (int i = 0; i < numData; i++)
438  {
439  _readBytes(4, &numObstructions);
440  _skip(numObstructions * 8);
441  }
442  int32_t visibilityMapSize[2];
443  _readBytes(8, visibilityMapSize);
444  _skip(visibilityMapSize[0] * visibilityMapSize[1] * 4);
445 }

894 int DefaultAnalyzer::_inflateRawHeader()
895 {
896  int ret;
897  unsigned have;
898  z_stream strm;
899  uint8_t in[ZLIB_CHUNK];
900  uint8_t out[ZLIB_CHUNK];
901 
902  _header.reserve(HEADER_INIT);
903 
904  /* allocate inflate state */
905  strm.zalloc = Z_NULL;
906  strm.zfree = Z_NULL;
907  strm.opaque = Z_NULL;
908  strm.avail_in = 0;
909  strm.next_in = Z_NULL;
910  ret = inflateInit2(&strm, -MAX_WBITS);
911  if (ret != Z_OK)
912  return ret;
913 
914  /* decompress until deflate stream ends or end of file */
915  do {
916  _f.read((char*)&in, ZLIB_CHUNK);
917  strm.avail_in = _f.gcount();
918  if (!_f.good()) {
919  (void)inflateEnd(&strm);
920  return Z_ERRNO;
921  }
922  if (strm.avail_in == 0)
923  break;
924  strm.next_in = in;
925 
926  /* run inflate() on input until output buffer not full */
927  do {
928  strm.avail_out = ZLIB_CHUNK;
929  strm.next_out = out;
930 
931  ret = inflate(&strm, Z_NO_FLUSH);
932  // assert(ret != Z_STREAM_ERROR); /* state not clobbered */
933  switch (ret) {
934  case Z_NEED_DICT:
935  ret = Z_DATA_ERROR; /* and fall through */
936  case Z_DATA_ERROR:
937  case Z_MEM_ERROR:
938  (void)inflateEnd(&strm);
939  return ret;
940  }
941 
942  have = ZLIB_CHUNK - strm.avail_out;
943  _header.insert(
944  _header.end(),
945  out,
946  out + have
947  );
948  if (!_f.good()) {
949  (void)inflateEnd(&strm);
950  return Z_ERRNO;
951  }
952  } while (strm.avail_out == 0);
953 
954  /* done when inflate() says it's done */
955  } while (ret != Z_STREAM_END);
956 
957  /* clean up and return */
958  (void)inflateEnd(&strm);
959  return ret == Z_STREAM_END ? Z_OK : Z_DATA_ERROR;
