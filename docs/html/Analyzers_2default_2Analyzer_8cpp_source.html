84 int DefaultAnalyzer::_inflateRawHeader()
85 {
86  int ret;
87  unsigned have;
88  z_stream strm;
89  uint8_t in[ZLIB_CHUNK];
90  uint8_t out[ZLIB_CHUNK];
91 
92  _header.reserve(HEADER_INIT);
93 
94  /* allocate inflate state */
95  strm.zalloc = Z_NULL;
96  strm.zfree = Z_NULL;
97  strm.opaque = Z_NULL;
98  strm.avail_in = 0;
99  strm.next_in = Z_NULL;
100  ret = inflateInit2(&strm, -MAX_WBITS);
101  if (ret != Z_OK)
102  return ret;
103 
104  /* decompress until deflate stream ends or end of file */
105  do {
106  _f.read((char*)&in, ZLIB_CHUNK);
107  strm.avail_in = _f.gcount();
108  if (!_f.good()) {
109  (void)inflateEnd(&strm);
110  return Z_ERRNO;
111  }
112  if (strm.avail_in == 0)
113  break;
114  strm.next_in = in;
115 
116  /* run inflate() on input until output buffer not full */
117  do {
118  strm.avail_out = ZLIB_CHUNK;
119  strm.next_out = out;
120 
121  ret = inflate(&strm, Z_NO_FLUSH);
122  // assert(ret != Z_STREAM_ERROR); /* state not clobbered */
123  switch (ret) {
124  case Z_NEED_DICT:
125  ret = Z_DATA_ERROR; /* and fall through */
126  case Z_DATA_ERROR:
127  case Z_MEM_ERROR:
128  (void)inflateEnd(&strm);
129  return ret;
130  }
131 
132  have = ZLIB_CHUNK - strm.avail_out;
133  _header.insert(
134  _header.end(),
135  out,
136  out + have
137  );
138  if (!_f.good()) {
139  (void)inflateEnd(&strm);
140  return Z_ERRNO;
141  }
142  } while (strm.avail_out == 0);
143 
144  /* done when inflate() says it's done */
145  } while (ret != Z_STREAM_END);
146 
147  /* clean up and return */
148  (void)inflateEnd(&strm);
149  return ret == Z_STREAM_END ? Z_OK : Z_DATA_ERROR;
150 }
