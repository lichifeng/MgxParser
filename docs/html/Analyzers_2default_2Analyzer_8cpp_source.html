58 bool DefaultAnalyzer::_locateStreams() {
59  /**
60  * \brief headerMeta[0]: header_len + next_pos + header_data (Data
61  * length) \n headerMeta[1]: nextPos
62  *
63  */
64  int32_t headerMeta[2];
65  _f.read((char*)headerMeta, 8);
66 
67  // Construct body stream
68  _bodySize = filesize - headerMeta[0];
69  _body.resize(_bodySize); /// \note vector 的 resize 和 reserve 是不一样的，这里因为这个问题卡了很久。reserve 并不会初始化空间，因此也不能直接读数据进去。
70  _f.seekg(headerMeta[0]);
71  _f.read((char*)_body.data(), _bodySize);
72 
73  uintmax_t rawHeaderPos = headerMeta[1] < filesize ? 8 : 4;
74  _f.seekg(rawHeaderPos);
75  if (rawHeaderPos == 4) versionCode = AOK;
76 
77  // Try to unzip header data
78  if (_inflateRawHeader() != 0) return false;
79 
80  // Start data analyzing
81  _analyze();
82 
83  return true;
84 }

252 int DefaultAnalyzer::_setVersionCode() {
253  /// \todo Every condition needs to be tested!
254  if (_bytecmp(versionStr, "TRL 9.3", 8)) {
255  return versionCode == AOK ? AOKTRIAL : AOCTRIAL;
256  }
257  if (_bytecmp(versionStr, "VER 9.3", 8)) return versionCode = AOK;
258  if (_bytecmp(versionStr, "VER 9.4", 8)) {
259  if (logVersion == 3) return versionCode = AOC10;
260  if (logVersion == 5 || saveVersion >= 12.9699) return versionCode = DE;
261  if (saveVersion >= 12.4999) return versionCode = HD50_6;
262  if (saveVersion >= 12.4899) return versionCode = HD48;
263  if (saveVersion >= 12.3599) return versionCode = HD46_7;
264  if (saveVersion >= 12.3399) return versionCode = HD43;
265  if (saveVersion > 11.7601) return versionCode = HD;
266  if (logVersion == 4) return versionCode = AOC10C;
267  return versionCode = AOC;
268  }
269  if (_bytecmp(versionStr, "VER 9.5", 8)) return versionCode = AOFE21;
270  if (_bytecmp(versionStr, "VER 9.8", 8)) return versionCode = USERPATCH12;
271  if (_bytecmp(versionStr, "VER 9.9", 8)) return versionCode = USERPATCH13;
272  if (_bytecmp(versionStr, "VER 9.A", 8)) return versionCode = USERPATCH14RC1;
273  if (_bytecmp(versionStr, "VER 9.B", 8)) return versionCode = USERPATCH14RC2;
274  if (_bytecmp(versionStr, "VER 9.C", 8) || _bytecmp(versionStr, "VER 9.D", 8)) return versionCode = USERPATCH14;
275  if (_bytecmp(versionStr, "VER 9.E", 8) || _bytecmp(versionStr, "VER 9.F", 8)) return versionCode = USERPATCH15;
276  if (_bytecmp(versionStr, "MCP 9.F", 8)) return versionCode = MCP;
277 
278  // If none above match:
279  status = "fail";
280  message.append("Detected unsupported game version.");
281  versionCode = UNSUPPORTED;
282  throw(AnalyzerException(message));
283 }

285 int DefaultAnalyzer::_inflateRawHeader()
286 {
287  int ret;
288  unsigned have;
289  z_stream strm;
290  uint8_t in[ZLIB_CHUNK];
291  uint8_t out[ZLIB_CHUNK];
292 
293  _header.reserve(HEADER_INIT);
294 
295  /* allocate inflate state */
296  strm.zalloc = Z_NULL;
297  strm.zfree = Z_NULL;
298  strm.opaque = Z_NULL;
299  strm.avail_in = 0;
300  strm.next_in = Z_NULL;
301  ret = inflateInit2(&strm, -MAX_WBITS);
302  if (ret != Z_OK)
303  return ret;
304 
305  /* decompress until deflate stream ends or end of file */
306  do {
307  _f.read((char*)&in, ZLIB_CHUNK);
308  strm.avail_in = _f.gcount();
309  if (!_f.good()) {
310  (void)inflateEnd(&strm);
311  return Z_ERRNO;
312  }
313  if (strm.avail_in == 0)
314  break;
315  strm.next_in = in;
316 
317  /* run inflate() on input until output buffer not full */
318  do {
319  strm.avail_out = ZLIB_CHUNK;
320  strm.next_out = out;
321 
322  ret = inflate(&strm, Z_NO_FLUSH);
323  // assert(ret != Z_STREAM_ERROR); /* state not clobbered */
324  switch (ret) {
325  case Z_NEED_DICT:
326  ret = Z_DATA_ERROR; /* and fall through */
327  case Z_DATA_ERROR:
328  case Z_MEM_ERROR:
329  (void)inflateEnd(&strm);
330  return ret;
331  }
332 
333  have = ZLIB_CHUNK - strm.avail_out;
334  _header.insert(
335  _header.end(),
336  out,
337  out + have
338  );
339  if (!_f.good()) {
340  (void)inflateEnd(&strm);
341  return Z_ERRNO;
342  }
343  } while (strm.avail_out == 0);
344 
345  /* done when inflate() says it's done */
346  } while (ret != Z_STREAM_END);
347 
348  /* clean up and return */
349  (void)inflateEnd(&strm);
350  return ret == Z_STREAM_END ? Z_OK : Z_DATA_ERROR;
351 }
