15 void DefaultAnalyzer::_findScenarioHeaderStart() {
16  // Try to locate scenario version data in DE (this float in DE varies
17  // among minor game versions and relative stable in previous versions.)
18  if (IS_DE(versionCode)) {
19  size_t searchSpan = 8000; // usually 5500~6500
20  auto startPoint = _victoryStartPos - searchSpan;
21  _curPos = startPoint;
22  for (size_t i = 0; i < searchSpan; i++)
23  {
24  scenarioVersion = *(float*)startPoint;
25  if (scenarioVersion > 1.35 && scenarioVersion < 1.55) {
26  _scenarioHeaderPos = _curPos = startPoint - 4; // 4 bytes are 00 00 00 00 before scenario version
27  message.append("[INFO] Reach _scenarioHeaderPos by scenario version range test. \n");
28  return;
29  }
30  ++startPoint;
31  }
32  throw(ParserException("[WARN] Cannot find satisfied _scenarioHeaderPos in this DE version. \n"));
33  }
34 
35 
36  auto scenarioSeprator = IS_AOK(versionCode) ? \
37  patterns::scenarioConstantAOK : \
38  (
39  IS_HD(versionCode) ?
40  (
41  /// \todo 这里应该是13.3399还是13.3599?
42  saveVersion >= 12.3599 ? \
43  patterns::scenarioConstantHD : \
44  patterns::scenarioConstantMGX2
45  ) :
46  patterns::scenarioConstantAOC
47  );
48 
49  vector<uint8_t>::reverse_iterator rFound;
50  rFound = findPosition(
51  make_reverse_iterator(_header.begin() + (_victoryStartPos - _header.data())),
52  _header.rend(),
53  scenarioSeprator.rbegin(),
54  scenarioSeprator.rend()
55  );
56  if (rFound == _header.rend()) {
57  /// \todo aoc-mgz says aok scenario version can appear in UP
58  throw(ParserException("[WARN] Cannot find satisfied _scenarioHeaderPos. \n"));
59  } else {
60  _scenarioHeaderPos = _curPos = &*(--rFound) - 4 - scenarioSeprator.size();
61  message.append("[INFO] Reach _scenarioHeaderPos and passed validation. \n");
62  }
63 }
