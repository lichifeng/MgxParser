46 /**
47  * @brief 用于编码转换的的工具，本质上是对 iconv 的一个封装。\n
48  * 使用前需要创建一个 EncodingConverter 类的实体，指定源编码和转换后的编码。
49  *
50  */
51 class EncodingConverter
52 {
53 public:
54  EncodingConverter();
55 
56  /**
57  * @brief Construct a new Encoding Converter object
58  *
59  * @param out_encode 转换后字符串的编码
60  * @param in_encode 转换前字符串的编码（通常会从录像文件中推断出来）
61  * @param ignore_error 忽略错误
62  * @param buf_size 缓冲区容量
63  */
64  EncodingConverter(const std::string &out_encode,
65  const std::string &in_encode,
66  bool ignore_error = false,
67  size_t buf_size = 1024)
68  : ignore_error_(ignore_error),
69  buf_size_(buf_size)
70  {
71  if (buf_size == 0)
72  {
73  throw std::runtime_error("buffer size must be greater than zero");
74  }
75 
76  iconv_t conv = ::iconv_open(out_encode.c_str(), in_encode.c_str());
77  if (conv == (iconv_t)-1)
78  {
79  if (errno == EINVAL)
80  throw std::runtime_error(
81  "not supported from " + in_encode + " to " + out_encode);
82  else
83  throw std::runtime_error("unknown error");
84  }
85  iconv_ = conv;
86  }
87 
88  ~EncodingConverter()
89  {
90  iconv_close(iconv_);
91  }
92 
93  /**
94  * @brief 主要方法，用来用行字符编码的转换
95  *
96  * @param input 源字符串（引用）
97  * @param output 转换后输出的字符串（引用）
98  */
99  void convert(const std::string &input, std::string &output) const
100  {
101  // copy the string to a buffer as iconv function requires a non-const char
102  // pointer.
103  std::vector<char> in_buf(input.begin(), input.end());
104  char *src_ptr = &in_buf[0];
105  size_t src_size = input.size();
106 
107  std::vector<char> buf(buf_size_);
108  std::string dst;
109  while (0 < src_size)
110  {
111  char *dst_ptr = &buf[0];
112  size_t dst_size = buf.size();
113  size_t res = ::iconv(iconv_, &src_ptr, &src_size, &dst_ptr, &dst_size);
114  if (res == (size_t)-1)
115  {
116  if (errno == E2BIG)
117  {
118  // ignore this error
119  }
120  else if (ignore_error_)
121  {
122  // skip character
123  ++src_ptr;
124  --src_size;
125  }
126  else
127  {
128  check_convert_error();
129  }
130  }
131  dst.append(&buf[0], buf.size() - dst_size);
132  }
133  dst.swap(output);
134  }
135 
136 private:
137  void check_convert_error() const
138  {
139  switch (errno)
140  {
141  case EILSEQ:
142  case EINVAL:
143  throw std::runtime_error("invalid multibyte chars");
144  default:
145  throw std::runtime_error("unknown error");
146  }
147  }
148 
149  iconv_t iconv_;
150  bool ignore_error_;
151  const size_t buf_size_;
152 };
