106 
107  // Init map
108  cimg_library::CImg<unsigned char> img(mapXY[0], mapXY[1], 1, 4, 0xff);
109 
110  // prepare raw data
111  if ((uint8_t)0xff == _mapBitmap[0])
112  {
113  versionOffset = 1;
114  versionSpan = 4;
115  }
116  else
117  {
118  versionOffset = 0;
119  versionSpan = 2;
120  }
121 
122  // populate bits
123  cimg_forXY(img, x, y)
124  {
125  curPos = (y * mapXY[0] + x) * versionSpan + versionOffset;
126  rbPos = ((y + 1) * mapXY[0] + x + 1) * versionSpan + versionOffset;
127 
128  elevation = 1;
129  if (x < mapXY[0] - 1 && y < mapXY[1] -1)
130  {
131  if (_mapBitmap[curPos + 1] > _mapBitmap[rbPos + 1])
132  elevation = 0;
133  else if (_mapBitmap[curPos + 1] < _mapBitmap[rbPos + 1])
134  elevation = 2;
135  }
136 
137  img(x, y, 0) = MAP_COLORS[_mapBitmap[curPos]][elevation][0];
138  img(x, y, 1) = MAP_COLORS[_mapBitmap[curPos]][elevation][1];
139  img(x, y, 2) = MAP_COLORS[_mapBitmap[curPos]][elevation][2];
140  img(x, y, 3) = 0xff;
141  }
142 
143  // draw objects
144  unsigned char RGB[4];
145  for (auto& obj : _gaiaObjs)
146  {
147  // NOTE: See this: https://stackoverflow.com/questions/5134614/c-const-map-element-access
148  RGB[0] = gaiaColors.at(obj.id).r;
149  RGB[1] = gaiaColors.at(obj.id).g;
150  RGB[2] = gaiaColors.at(obj.id).b;

153  }

162  }

217 int Record::_inflateRawHeader()
218 {
219  int ret;
220  unsigned have;
221  z_stream strm;
222  unsigned char in[ZLIB_CHUNK];
223  unsigned char out[ZLIB_CHUNK];
224 
225  _header.reserve(HEADER_INIT);
226 
227  /* allocate inflate state */
228  strm.zalloc = Z_NULL;
229  strm.zfree = Z_NULL;
230  strm.opaque = Z_NULL;
231  strm.avail_in = 0;
232  strm.next_in = Z_NULL;
233  ret = inflateInit2(&strm, -MAX_WBITS);
234  if (ret != Z_OK)
235  return ret;
236 
237  /* decompress until deflate stream ends or end of file */
238  do {
239  _file->read((char*)&in, ZLIB_CHUNK);
240  strm.avail_in = _file->gcount();
241  if (!_file->good()) {
242  (void)inflateEnd(&strm);
243  return Z_ERRNO;
244  }
245  if (strm.avail_in == 0)
246  break;
247  strm.next_in = in;
248 
249  /* run inflate() on input until output buffer not full */
250  do {
251  strm.avail_out = ZLIB_CHUNK;
252  strm.next_out = out;
253 
254  ret = inflate(&strm, Z_NO_FLUSH);
255  // assert(ret != Z_STREAM_ERROR); /* state not clobbered */
256  switch (ret) {
257  case Z_NEED_DICT:
258  ret = Z_DATA_ERROR; /* and fall through */
259  case Z_DATA_ERROR:
260  case Z_MEM_ERROR:
261  (void)inflateEnd(&strm);
262  return ret;
263  }
264 
265  have = ZLIB_CHUNK - strm.avail_out;
266  _header.insert(
267  _header.end(),
268  out,
269  out + have
270  );
271  if (!_file->good()) {
272  (void)inflateEnd(&strm);
273  return Z_ERRNO;
274  }
275  } while (strm.avail_out == 0);
276 
277  /* done when inflate() says it's done */
278  } while (ret != Z_STREAM_END);
279 
280  /* clean up and return */
281  (void)inflateEnd(&strm);
282  return ret == Z_STREAM_END ? Z_OK : Z_DATA_ERROR;
283 }
